  defp assert_eventually(fun, timeout_ms \\ 5_000, interval_ms \\ 100) do
    deadline = System.monotonic_time(:millisecond) + timeout_ms
    do_assert_eventually(fun, deadline, interval_ms)
  end

  defp do_assert_eventually(fun, deadline, interval_ms) do
    if fun.() do
      true
    else
      if System.monotonic_time(:millisecond) >= deadline do
        flunk("Assertion did not become true within timeout")
      else
        Process.sleep(interval_ms)
        do_assert_eventually(fun, deadline, interval_ms)
      end
    end
  end
end

defmodule Slackex.Chat.MessageGrouping do
  @moduledoc """
  Pure functions for annotating messages with grouping and time divider metadata.

  A message is "grouped" (compact display) when it has the same sender as the
  previous message, is within 5 minutes, and neither is deleted or a thread reply.

  Time dividers are shown when there's a 30+ minute gap between messages.
  """

  alias Slackex.Chat.Message

  @group_window_seconds 5 * 60
  @divider_gap_seconds 30 * 60

  @doc """
  Annotates a list of messages with `:grouped`, `:show_divider`, and `:divider_label` fields.
  """
  @spec annotate([Message.t()]) :: [Message.t()]
  def annotate([]), do: []

  def annotate([first | rest]) do
    first = %{first | grouped: false, show_divider: false, divider_label: nil}

    {annotated, _prev} =
      Enum.map_reduce(rest, first, fn msg, prev ->
        grouped = should_group?(msg, prev)
        {show_divider, divider_label} = divider_info(msg, prev)

        annotated = %{msg | grouped: grouped, show_divider: show_divider, divider_label: divider_label}
        {annotated, msg}
      end)

    [first | annotated]
  end

  @doc """
  Determines if `incoming` should be grouped with `last_message`.
  Used for incremental PubSub updates where only the last message is known.
  """
  @spec should_group?(Message.t(), Message.t() | nil) :: boolean()
  def should_group?(_incoming, nil), do: false

  def should_group?(incoming, last) do
    same_sender?(incoming, last) &&
      within_window?(incoming, last) &&
      not deleted?(last) &&
      not thread_reply?(incoming)
  end

  @doc """
  Determines divider info for `incoming` compared to `last_message`.
  Returns `{show_divider, divider_label}`.
  """
  @spec divider_info(Message.t(), Message.t() | nil) :: {boolean(), String.t() | nil}
  def divider_info(_incoming, nil), do: {false, nil}

  def divider_info(incoming, last) do
    gap = DateTime.diff(incoming.inserted_at, last.inserted_at, :second)

    if gap >= @divider_gap_seconds do
      {true, format_divider_label(incoming.inserted_at)}
    else
      {false, nil}
    end
  end

  defp same_sender?(a, b), do: a.sender_id == b.sender_id

  defp within_window?(incoming, last) do
    DateTime.diff(incoming.inserted_at, last.inserted_at, :second) <= @group_window_seconds
  end

  defp deleted?(msg), do: msg.deleted_at != nil

  defp thread_reply?(msg), do: msg.parent_message_id != nil

  defp format_divider_label(datetime) do
    today = DateTime.utc_now() |> DateTime.to_date()
    msg_date = DateTime.to_date(datetime)
    time_str = Calendar.strftime(datetime, "%H:%M")

    cond do
      msg_date == today ->
        "Today at #{time_str}"

      msg_date == Date.add(today, -1) ->
        "Yesterday at #{time_str}"

      true ->
        Calendar.strftime(datetime, "%B %-d at %H:%M")
    end
  end
end
